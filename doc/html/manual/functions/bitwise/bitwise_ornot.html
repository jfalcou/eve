bitwise_ornot¶

Required header #include <eve/function/bitwise_ornot.hpp>

namespace eve
{
  constexpr Value0 bitwise_ornot( Value0 lhs, Value1 rhs ) noexcept
}

Function object for performing bitwise ornot over two Values of possibly different types but of same bit size, (lhs | ~rhs).

The result type is the one of the first operator.

Parameters¶

  • lhs, rhs : input values

Return value¶

The bitwise or of lhs and the complement of rhs for every elements of each parameter.

Options¶

Example¶

#include <eve/function/bitwise.hpp>
#include <eve/constant/nan.hpp>
#include <eve/wide.hpp>
#include <iostream>

using wide_it = eve::wide<std::uint32_t, eve::fixed<4>>;

int main() {
  wide_it pi = {3, 5, 0, 127};
  wide_it qi = {4, 1, 0, 1023};
  std::cout << "---- simd" << '\n'
            << " <- pi =                        " << pi << '\n'
            << " <- qi =                        " << qi << '\n'
            << " -> eve::bitwise_ornot(pi, qi) = " << eve::bitwise_ornot(pi, qi)
            << '\n';

  std::uint32_t xi = 3, yi = 4;

  std::cout << "---- scalar" << '\n'
            << " xi =                           " << xi << '\n'
            << " yi =                           " << yi << '\n'
            << " -> eve::bitwise_ornot(xi, yi) = " << eve::bitwise_ornot(xi, yi)
            << '\n';
  return 0;
}

Possible output

---- simd
 <- pi =                        (3, 5, 0, 127)
 <- qi =                        (4, 1, 0, 1023)
 -> eve::bitwise_ornot(pi, qi) = (4294967291, 4294967295, 4294967295, 4294966399)
---- scalar
 xi =                           3
 yi =                           4
 -> eve::bitwise_ornot(xi, yi) = 4294967291
