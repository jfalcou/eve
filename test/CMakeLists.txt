if(Boost_FOUND)
  set(EVE_USE_BOOST 1)

  string(REPLACE "/usr/include" "" Boost_INCLUDE_DIRS ${Boost_INCLUDE_DIRS})
else()
  set(Boost_INCLUDE_DIRS "")
endif()
