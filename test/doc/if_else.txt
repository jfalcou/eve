---- simd
 <- si =                              (2, -3, 0, 4)
 <- lsi =                             (true, false, false, true)
 <- pi =                              (3, -2, -10, 0)
 <- qi =                              (4, -1, 0, 5)
 -> eve::if_else(si, pi, qi) =        (3, -2, 0, 0)
 -> eve::if_else(lsi, pi, qi) =       (3, -1, 0, 0)
---- scalar
 ssi =                                3
 lssi =                               false
 xi =                                 3
 yi =                                 4
 -> eve::if_else(ssi, xi, yi) =       3
 -> eve::if_else(lssi, xi, yi) =      4
