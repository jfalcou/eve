---- simd
 <- si =                              (2, -3, 0, 4)
 <- lsi =                             (true, false, false, true)
 <- pi =                              (3, -2, -10, 0)
 -> eve::if_else_zero(si, pi, eve::zero_) =        (3, -2, 0, 0)
 -> eve::if_else(lsi, pi, eve::zero_) =       (3, 0, 0, 0)
---- scalar
 ssi =                                3
 lssi =                               false
 xi =                                 3
 -> eve::if_else(ssi, xi, eve::zero_) =       3
 -> eve::if_else(lssi, xi, eve::zero_) =      0
