---- simd
 <- si =                              (2, -3, 0, 4)
 <- lsi =                             (true, false, false, true)
 <- pi =                              (3, -2, -10, 0)
 -> eve::if_else(si, pi, eve::allbits_) =        (3, -2, -1, 0)
 -> eve::if_else(lsi, pi, eve::allbits_) =       (3, -1, -1, 0)
---- scalar
 ssi =                                3
 lssi =                               false
 xi =                                 3
 -> eve::if_else(ssi, xi, eve::allbits_) =       3
 -> eve::if_else(lssi, xi, eve::allbits_) =      -1
