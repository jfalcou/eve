---- simd
 <- si =                              (2, -3, 0, 4)
 <- lsi =                             (true, false, false, true)
 <- pi =                              (3, -2, -10, 0)
 -> eve::if_else(si, eve::allbits_, pi) =        (-1, -1, -10, -1)
 -> eve::if_else(lsi, eve::allbits, pi_) =       (-1, -2, -10, -1)
---- scalar
 ssi =                                3
 lssi =                               false
 xi =                                 3
 -> eve::if_else(ssi, eve::allbits_, xi) =       -1
 -> eve::if_else(lssi, xi, eve::allbits_, xi) =      3
